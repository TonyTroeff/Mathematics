\documentclass[tikz, border=10pt]{standalone}

\usepackage{tikz}
\usepackage{tikz-3dplot}
\usetikzlibrary{calc}

\begin{document}

\tdplotsetmaincoords{60}{125}

\begin{tikzpicture}[tdplot_main_coords ]

    \coordinate (A) at (3,-3,-3);
    \coordinate (B) at (-3,3,-3);
    \coordinate (C) at (3,3,3);
    \coordinate (D) at (-3,-3,3);

    % M = centroid of triangle ACD
    \coordinate (M) at ($(A) !1/2! (C) !1/3! (D)$);
    
    % N = midpoint of BC
    \coordinate (N) at ($(B) !1/2! (C)$);

    % Q = midpoint of AC
    \coordinate (Q) at ($(A) !1/2! (C)$);

    \draw[black] (A) to (B);
    \draw[black] (B) to (D);	
    \draw[black] (D) to (A);
    \draw[black, dashed] (A) to (C);
    \draw[black, dashed] (B) to (C);
    \draw[black, dashed] (D) to (C);
    \draw[black, dashed] (D) to (Q);

    \fill[red!10, opacity=0.5] (A) to (C) to (D) to cycle;
    \fill[blue] (M) circle (1pt);
    \fill[blue] (N) circle (1pt);
    \fill[black] (Q) circle (1pt);

    % Labels
    \node[left] at (A) {$A$};
    \node[right] at (B) {$B$};
    \node[above right] at (C) {$C$};
    \node[above] at (D) {$D$};
    \node[above right] at (M) {$M$};
    \node[above right] at (N) {$N$};
    \node[below] at (Q) {$Q$};

    % Draw vectors
    \draw[->, thick, blue] (B) to (M);
    \draw[->, thick, blue] (D) to (N);

\end{tikzpicture}

\end{document}
